// Arrays
      int[] my_array = new int[]{1, 3, 4,  6, 7, 8};
      my_array[0];      // equivalent to 1
      my_array[2];       // equivalent to 4
      
      my_array[2] = 5;
      my_array;          // equivalent to  [1, 3, 5,  6, 7, 8]
---------------------------------------      
      // Lists
      List<Integer> my_list = new ArrayList<Integer>();    //starts as an empty list
      my_list.add(3);    // add the integer 3 to my_list
      my_list.add(5);    // add the integer 5 to my_list
      my_list;           // equivalent to (3, 5)
      my_list.get(1);    // equivalent to 5
      my_list.set(1, 7); //reassign the contents of the 1 index
      my_list;           // equivalent to (3, 7)
