// iterative for loop
      for (int i = 0; i < 10; i++){
         System.out.println(i); // will iteratively output all numbers from 0 to 9 inclusive
      }
---------------------------------------      
      // for each loop
      int[] my_array = new int[]{1, 3, 5, 6, 4 ,2};
      for (Integer i : my_array){
         System.out.println(i); // will print all elements in my_array starting from 0th index
      }
---------------------------------------      
      // while loop
      int i = 0;
      while (i  < 10){
         System.out.println(i); //prints out all numbers from 0 to 9 inclusive
         i = i + 1;
      }
